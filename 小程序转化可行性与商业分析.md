### Step 3：JavaScript逻辑（8小时）

```javascript
// pages/main/main.js
const apiBase = 'https://web-production-3853a.up.railway.app'

Page({
  data: {
    sessionId: '',
    sessionName: '',
    myName: 'User_' + Math.random().toString(36).substr(2, 5).toUpperCase(),
    headerBg: '#FF6B35',
    tab: 'edit',
    dates: [],
    hours: [],
    participants: [],
    heatmapData: [],
    myAvail: {},
    loading: false
  },

  onLoad(options) {
    // 从分享参数获取sessionId
    const sessionId = options.s || ''
    this.setData({ sessionId })
    this.loadSession()
  },

  loadSession() {
    const { sessionId } = this.data
    wx.showLoading({ title: '加载中...' })
    
    wx.request({
      url: `${apiBase}/api/session/${sessionId}`,
      method: 'GET',
      success: (res) => {
        if (res.statusCode === 200) {
          this.initializeData(res.data)
        } else {
          wx.showToast({ title: '会话不存在', icon: 'error' })
        }
      },
      fail: () => {
        wx.showToast({ title: '网络错误', icon: 'error' })
      },
      complete: () => wx.hideLoading()
    })
  },

  initializeData(session) {
    const dates = this.generateDates(session.dateS, session.dateE)
    const hours = Array.from(
      {length: session.hourE - session.hourS},
      (_, i) => session.hourS + i
    )
    
    this.setData({
      sessionName: session.name,
      participants: session.participants,
      dates,
      hours,
      myAvail: this.extractMyAvail(session)
    })
    
    this.renderHeatmap(session)
  },

  toggleCell(e) {
    const { date, hour } = e.currentTarget.dataset
    const h = parseInt(hour)
    
    let avail = this.data.myAvail[date] || []
    const idx = avail.indexOf(h)
    
    if (idx > -1) {
      avail.splice(idx, 1)
    } else {
      avail.push(h)
      avail.sort((a, b) => a - b)
    }
    
    const newAvail = { ...this.data.myAvail, [date]: avail }
    this.setData({ myAvail: newAvail })
    
    // 上传更新
    this.uploadAvail(newAvail)
  },

  uploadAvail(avail) {
    const { sessionId, myName } = this.data
    
    wx.request({
      url: `${apiBase}/api/session/${sessionId}/avail`,
      method: 'PUT',
      data: { name: myName, avail },
      success: () => {
        this.loadSession() // 刷新显示
      }
    })
  },

  generateDates(dateS, dateE) {
    const dates = []
    const start = new Date(dateS)
    const end = new Date(dateE)
    
    for (let d = new Date(start); d <= end; d.setDate(d.getDate() + 1)) {
      dates.push({
        date: d.toISOString().split('T')[0],
        label: d.toLocaleDateString('zh-CN', {
          month: '2-digit',
          day: '2-digit',
          weekday: 'short'
        })
      })
    }
    return dates
  },

  renderHeatmap(session) {
    const heatmapData = []
    
    for (const date of this.data.dates) {
      for (const hour of this.data.hours) {
        const people = session.participants
          .filter(p => p.avail[date.date]?.includes(hour))
          .map(p => ({ name: p.name, color: p.color }))
        
        heatmapData.push({
          time: `${date.label} ${hour}:00`,
          count: people.length,
          people
        })
      }
    }
    
    this.setData({ heatmapData })
  },

  switchTab(e) {
    this.setData({ tab: e.currentTarget.dataset.tab })
  },

  onShareAppMessage() {
    return {
      title: this.data.sessionName || '时间调查',
      path: `/pages/main/main?s=${this.data.sessionId}`
    }
  },

  hasAvail(date, hour) {
    return this.data.myAvail[date]?.includes(parseInt(hour)) || false
  },

  extractMyAvail(session) {
    const myName = this.data.myName
    const me = session.participants.find(p => p.name === myName)
    return me?.avail || {}
  }
})
```
